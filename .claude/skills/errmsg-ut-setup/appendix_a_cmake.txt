# 附录 A：CMakeLists.txt 追加内容

## 位置
在 `tests/ut/runtime/runtime/CMakeLists.txt` 文件末尾追加。

## 内容

```cmake
# ========== ErrMsg 真实实现验证测试目标 ==========
# 此目标链接真实 ErrorManager，用于验证 ErrMsg 格式化输出
# 仅用于本地验证，不提交到仓库
###################################################################################################
add_executable(runtime_utest_errmsg_real
    ${runtime_raw_device_adpt_common_list}
    ${TOP_DIR}/src/runtime/driver/npu_driver.cc
    ${TOP_DIR}/src/runtime/driver/npu_driver_mem.cc
    ${TOP_DIR}/src/runtime/driver/npu_driver_queue.cc
    ${TOP_DIR}/src/runtime/driver/npu_driver_res.cc
    ${TOP_DIR}/src/runtime/driver/npu_driver_dcache_lock_opb.cpp
    ${TOP_DIR}/src/runtime/driver/npu_driver_standard_soc.cc
    ${TOP_DIR}/src/runtime/driver/v100/npu_driver.cc
    ${TOP_DIR}/src/runtime/core/src/kernel/v100/kernel.cc
    ${TOP_DIR}/src/runtime/core/src/task/v100/stub_task.cc
    ${TOP_DIR}/src/runtime/core/src/launch/xpu_aicpu_c_stub.cc
    
    # 真实 ErrorManager 实现（关键：替代 stub）
    ${TOP_DIR}/src/dfx/error_manager/error_manager.cc
    
    # 只保留必要的 stub（排除 error_manager_stub.cc）
    stub/hal_stub.cc
    stub/atrace_stub.cc
    stub/awatch_dog_stub.cc
    stub/log_stub.cc
    stub/prof_stub.cc
    stub/platform_stub.cc
    
    # 测试文件
    test/main.cc
    test/rt_errmsg_real_test.cc
)

target_include_directories(runtime_utest_errmsg_real PRIVATE
    ${TOP_DIR}/src/dfx/error_manager
    ${TOP_DIR}/include/dfx
    ${TOP_DIR}/include/dfx/base
    ${TOP_DIR}/src/runtime/core/inc
    ${TOP_DIR}/src/runtime/core/inc/common
    ${TOP_DIR}/src/inc
    ${CANN_3RD_LIB_PATH}/mockcpp_src/include
)

target_compile_definitions(runtime_utest_errmsg_real PUBLIC
    TEMP_PERFORMANCE
    TEMP_PERFORMANCE_NOT_OUTPUT
)

# 关键：取消 CFG_DEV_PLATFORM_PC 定义，让 ErrMsg 宏调用真实 ErrorManager
target_compile_options(runtime_utest_errmsg_real PRIVATE
    -UCFG_DEV_PLATFORM_PC
)

target_link_libraries(runtime_utest_errmsg_real PRIVATE
    $<BUILD_INTERFACE:runtime_utest_intf_pub>
    -Wl,--whole-archive
    runtimeut_src_static_lib
    -Wl,--no-whole-archive
    ${CANN_3RD_LIB_PATH}/mockcpp/lib/libmockcpp.a
    pthread
    dl
)

# 将新目标添加到依赖列表
add_dependencies(runtime_ut_all runtime_utest_errmsg_real)
```

## 关键配置说明

| 配置项 | 说明 |
|--------|------|
| `${TOP_DIR}/src/dfx/error_manager/error_manager.cc` | 真实 ErrorManager 实现，替代 stub |
| `stub/hal_stub.cc` | 必须保留，提供 hal 函数 stub |
| **不包含** `stub/error_manager_stub.cc` | 排除 stub，使用真实实现 |
| `-UCFG_DEV_PLATFORM_PC` | 取消定义，让宏调用真实 ErrorManager |
