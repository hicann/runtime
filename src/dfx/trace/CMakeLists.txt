    -Werror
    -Wall
    -Wextra
    -Wundef
    -Wunused
    -Wpointer-arith
    -Wdate-time
    -Wfloat-equal
    -Wformat=2
    -Wshadow
    -Wsign-compare
    -Wvla
    -Wdisabled-optimization
    -Wempty-body
    -Wignored-qualifiers
    -Wtype-limits
    -Wshift-negative-value

    -Wwrite-strings
    -Wmissing-format-attribute
    -Wformat-nonliteral

    $<BUILD_INTERFACE:${ATRACE_DIR}/../../../pkg_inc/watchdog>

    $<INSTALL_INTERFACE:include/awatchdog>
)
